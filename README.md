# NAME

XML::CompileX::WSDL11::AsMethods - Export SOAP operations as Perl methods

# VERSION

version 0.001

# SYNOPSIS

    use XML::CompileX::WSDL11::AsMethods;

    XML::CompileX::WSDL11::AsMethods->export('foo.wsdl');

# SUPPORT
